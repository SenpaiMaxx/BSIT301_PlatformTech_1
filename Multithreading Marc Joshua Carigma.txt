#include <iostream>
#include <thread>
#include <string>

using namespace std;

void one(int x){
	int i = 0;
	for(i=0; i<x; i++){
		cout << "This is thread No. 1  \n";
	}
}

void two(int x){
	int i = 0;
	for(i=0; i<x; i++){
		cout << "This is thread No. 2  \n";
	}
}

void three(int x){
	int i = 0;
	for(i=0; i<x; i++){
		cout << "This is thread No. 3 \n";
	}
}

void four(int x){
	int i = 0;
	for(i=0; i<x; i++){
		cout << "This is thread No. 4 \n";
	}
}

int main(){
	
	int i = 0;
	thread td1 (one, 3);
	thread td2 (two, 3);
	thread td3 (three, 3);
	thread td4 (four, 3);
	
	for(i=0; i<x; i++){
		cout << "This is from main \n";
	}
	
	td1.join();
	td2.join();
	td3.join();
	td4.join();
	return 0;
}
